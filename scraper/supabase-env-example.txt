# Supabase Configuration
# Copy these variables to your .env file and fill in your actual Supabase credentials

# Your Supabase project URL
# Find this in your Supabase dashboard under Settings > API
SUPABASE_URL=https://your-project-id.supabase.co

# Your Supabase anon/public key  
# This is safe to use in client-side code
# Find this in your Supabase dashboard under Settings > API
SUPABASE_ANON_KEY=your-anon-key-here

# Your Supabase service role key (KEEP SECRET!)
# This provides admin access - never expose in client-side code
# Find this in your Supabase dashboard under Settings > API
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here

# Optional: Environment setting
NODE_ENV=development

# Optional: Enable debug logging
DEBUG=supabase:*
